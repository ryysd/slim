static inline BOOL mc_vec_states_valid(Vector *v) {
  unsigned int i, j;
  for (i = 0, j = 1; i < vec_num(v) && j < vec_num(v); i++, j++) {
    State *s, *t;
    s = (State *)vec_get(v, i);
    t = (State *)vec_get(v, j);
    if (!state_succ_contains(s, t)) return FALSE;
  }

  return TRUE;
}
