#include <stdarg.h>
#include <stdlib.h>
#include "lmntal.h"
#include "error.h"

void do_lmn_fatal(const char *file, int line, const char *msg)
{
  fprintf(stderr,"%s(%d) %s\n", file, line, msg);
  fflush(stderr);
}

void lmn_report(const char *msg, ...)
{
  va_list args;
  va_start(args, msg);
  /* use raw port */
  vfprintf(stderr, msg, args);
  va_end(args);
  fputc('\n', stderr);
  fflush(stderr);
}
